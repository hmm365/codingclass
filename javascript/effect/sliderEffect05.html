const sliderWrap = document.querySelector(".slider__wrap");
const sliderImg = document.querySelector(".slider__img"); //보여지는영역
const sliderInner = document.querySelector(".slider__inner"); //움직이는 영역
const slider = document.querySelectorAll(".slider"); //각각 이미지
const sliderDot = document.querySelector(".slider__dot");
let currentIndex = 0; //현재 이미지
let dotIndex = ""; // 닷
let sliderCount = slider.length; //이미지갯수
let sliderWidth = sliderImg.offsetWidth; //이미지 가로값
// 닷만들기
function init() {
    slider.forEach((e, i) => (dotIndex += "<a href='#'' class='dot'>이미지" + (i + 1) + "</a>"));
    sliderDot.innerHTML = dotIndex;
    //첫번째 닷버튼한테 활성화표시
    sliderDot.firstChild.classList.add("active");
}
init();
//이미지 움직이는 영역
function gotoSlider(num) {
    sliderInner.style.transition = "all 400ms";
    sliderInner.style.transform = "translateX(" + -sliderWidth * num + "px)";
    let sliderDotActive = document.querySelectorAll(".slider__dot .dot");
    sliderDotActive.forEach((el) => el.classList.remove("active"));
    sliderDot.children[num].classList.add("active");
    currentIndex = num;
}
//버튼클릭했을때
document.querySelectorAll(".slider__btn a").forEach((btn, index) => {
    btn.addEventListener("click", () => {
        let prevIndex = (currentIndex + sliderCount - 1) % sliderCount;
        let nextIndex = (currentIndex + 1) % sliderCount;
        if (btn.classList.contains("prev")) {
            gotoSlider(prevIndex);
        } else {
            gotoSlider(nextIndex);
        }
    });
});
//닷버튼 클릭했을때
document.querySelectorAll(".slider__dot .dot").forEach((e, i) => {
    e.addEventListener("click", () => {
        gotoSlider(i);
    });
});
