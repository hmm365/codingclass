const sliderInner = document.querySelector(".slider__inner");
const slider = document.querySelectorAll(".slider");
let currentIndex = 0;

sliderInner.style.transition = "all 0.6s" // 애니메이션 효과
setInterval(() => {
    sliderInner.style.transform = `translateX(${-800 * currentIndex}px)`;   // 위치 변경
    currentIndex = (1 + currentIndex) % slider.length; // 위치변경할 값
}, 1000 ); 

let currentIndex = 0;
    const slider = document.querySelectorAll(".slider");
    setInterval(() => {
        currentIndex = (1 + currentIndex) % slider.length; // 위치 변경 할 값
        gsap.to(".slider__inner", { //선택자
            duration : 0.6, //초
            x : -800 * currentIndex //위치변경
        });
    }, 1000);

let currentIndex = 0;
    setInterval(() => {
        currentIndex = (currentIndex + 1) % $(".slider").length; //위치 변경 할 값
        $(".slider__inner").css("position", "relative"); // slider__inner position을 relative 변경
        $(".slider__inner").animate({left : -800 * currentIndex},600) // slider__inner left 만큼 이동
    }, 1000);
