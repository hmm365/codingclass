    let cook = ['당근','소세지','삼겹살','계란후라이']
    for(sel=0; sel < cook.length; sel++){
        if(cook[sel]!=='당근'){
            let i = '오늘 저녁은'+cook[sel];
            console.log(i);
        }
    }
}
