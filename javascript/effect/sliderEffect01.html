const sliderWrap = document.querySelector(".slider__wrap");
const sliderImg = document.querySelector(".slider__img");
const slider = document.querySelectorAll(".slider");
let timeCount = 0;
let currentIndex = 0;                   //현재 보이는 이미지
let sliderCount = slider.length;        //이미지 갯수
const silderTimer = setInterval(() => {
    // timeCount == sliderCount * 5 -1 ? clearInterval(silderTimer) : timeCount++;
    timeCount == timeCount++
    //console.log(timeCount)
    let nextIndex = (currentIndex + 1 ) % sliderCount;
    slider[currentIndex].style.opacity = "0"; //첫번째 이미지를 안보이게
    slider[nextIndex].style.opacity = "1"; //두번째 이미지를 보이게
    currentIndex = nextIndex; //currentIndex를 다음 이미지로 넘김
}, 1000);
