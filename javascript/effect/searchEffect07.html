const searchTime = document.querySelector(".time span");
const searchList = document.querySelector(".search__list");
const searchAnswers = document.querySelector(".search__answers");
const searchMissAnswers = document.querySelector(".search__missAnswers");

const searchStart = document.querySelector(".search__box .start");
const searchInput = document.querySelector("#search");
const searchAudio = document.querySelector("#audio");

const searchAudioPlay = document.querySelector("#play");
const searchAudioStop = document.querySelector("#stop");

const searchScoreTotal = document.querySelector(".search__info .count");
const searchScoreNow = document.querySelector(".search__info .num");

const searchResult = document.querySelector(".search__result .result");
const searchResultWrap = document.querySelector(".search__result");
const searchRestart = document.querySelector(".search__result .restart");

const bgm = new Audio("../assets/audio/Alert 1.m4a");
const bgm2 = new Audio("../assets/audio/Notification 1.m4a");

let timeReamining = 4, //남은 시간
    score = 0, //점수
    timeInterval = "",
    answers = {}; //새로운 정답

function upDateList() {
    cssProperty.forEach((data) => {
        searchList.innerHTML += `<span>${data.name}</span>`;
    });
}
upDateList();

function audioPlay() {
    searchAudio.volume = 0.2;
    searchAudio.play();
    searchAudioPlay.style.display = "none";
    searchAudioStop.style.display = "inline-block";
}
function audioStop() {
    searchAudio.pause();
    searchAudioStop.style.display = "none";
    searchAudioPlay.style.display = "inline-block";
}

//게임 시작하기
function startQuiz() {
    // 시작버튼 없애기
    searchStart.style.display = "none";
    searchList.style.display = "none";
    // 다시 시작할 때 기존 데이터 초기화
    searchAnswers.innerHTML = "";
    searchMissAnswers.innerHTML = "";
    //시간 설정
    timeInterval = setInterval(reduceTime, 1000);
    //뮤직 추가하기
    audioPlay();
    // 점수 계산
    searchScoreTotal.innerText = cssProperty.length;
    //정답체크
    checkAnswers();
    //게임끝
}

//인풋 체크
function checkInput() {
    let input = event.currentTarget.value.trim().toLocaleLowerCase();

    if (answers.hasOwnProperty(input) && !answers[input]) {
        answers[input] = true;

        //정답 표시하기
        searchAnswers.style.display = "block";
        searchAnswers.innerHTML += `<span>${input}</span>`;
        bgm.play();

        //점수 반영
        score++;
        searchScoreNow.innerText = score;
        //정답 초기화
        searchInput.value = "";
    } else {
        bgm2.play();
    }
    // console.log(answers);
}

//정답 체크하기
function checkAnswers() {
    cssProperty.forEach((item) => {
        let answer = item.name.trim().toLocaleLowerCase();
        answers[answer] = false;
    });
    // console.log(answers);
}

//오답 보여주기
function missAnswers() {
    searchMissAnswers.style.display = "block";

    cssProperty.forEach((item) => {
        let answer = item.name.trim().toLocaleLowerCase();
        if (!answers[answer]) {
            searchMissAnswers.innerHTML += `<span>${item.name}</span>`;
        }
    });
}

//시간 설정하기
function reduceTime() {
    timeReamining--;
    if (timeReamining == 0) endQuiz();
    searchTime.innerText = displayTime();
}
//시간 표시하기
function displayTime() {
    if (timeReamining <= 0) {
        return "0:00";
    } else {
        let minutes = Math.floor(timeReamining / 60);
        let seconds = timeReamining % 60;

        //초 단위가 한자리수 일 때 0을 추가
        seconds < 10 ? (seconds = "0" + seconds) : seconds;
        return `${minutes}:${seconds}`;
    }
}
//게임 끝났을때
function endQuiz() {
    //시작 버튼 만들기
    searchStart.style.display = "block";
    searchStart.style.pointerEvents = "none"; //버튼 누르지 못하게 하기

    //오답 보여주기
    missAnswers();

    //음악끄기
    audioStop();

    //시간 정지
    clearInterval(timeInterval);

    //결과값
    searchResultWrap.classList.add("show");
    let point = Math.round((score / cssProperty.length) * 100);
    searchResult.innerHTML = `게임이 끝났습니다.<br> 당신은 ${cssProperty.length}개 중에 ${score}개 를 맞추었습니다. <br> 당신의 점수는 ${point}입니다.`;
}
// 다시 시작하기
function restart() {
    searchResultWrap.classList.remove("show");

    timeReamining = 120;
    score = 0;
    searchScoreNow.innerText = "0";

    searchAudio.play();

    startQuiz();
}
//버튼 이벤트
searchStart.addEventListener("click", () => {
    startQuiz();
});
searchInput.addEventListener("input", () => {
    checkInput();
});
searchRestart.addEventListener("click", () => {
    restart();
});
//노래 시작
searchAudioPlay.addEventListener("click", () => {
    audioPlay();
});
//노래 정지
searchAudioStop.addEventListener("click", () => {
    audioStop();
});
