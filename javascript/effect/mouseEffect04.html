const cursor = document.querySelector(".mouse__cursor");
const cursorRect = cursor.getBoundingClientRect();
document.querySelector(".mouse__wrap figure").addEventListener("mousemove", (e) => {
    //커서
    gsap.to(cursor, {
        duration: 0.5,
        left: e.pageX - cursorRect.width / 2,
        top: e.pageY - cursorRect.height / 2,
    });
    //마우스 좌표값
    let mousePageX = e.pageX;
    let mousePageY = e.pageY;
    //전체 가로
    // window.innerWidth; // 브라우저 크기
    // window.outerWidth; // 브라우저 크기(스크롤바 포함,관리자 도구 포함)
    // window.screen.width; // 화면 크기
    // 마우스 좌표 값 가운데로 초기화
    // 전체 길이/2 0 마우스 좌표값 == 0
    let centerPageX = window.innerWidth / 2 - mousePageX;
    let centerPageY = window.innerHeight / 2 - mousePageY;
    //이미지 움직임
    const imgMove = document.querySelector(".mouse__wrap figure img");
    // imgMove.style.transform = "translate(" + centerPageX / 20 + "px, " + centerPageY / 20 + "px)";
    gsap.to(imgMove, {
        duration: 0.5,
        left: centerPageX / 20,
        top: centerPageY / 20,
    });
    //출력
    document.querySelector(".mousePageX").textContent = mousePageX;
    document.querySelector(".mousePageY").textContent = mousePageY;
    document.querySelector(".centerPageX").textContent = centerPageX;
    document.querySelector(".centerPageY").textContent = centerPageY;
});
