const number4 = [100, 200, 300, 400, 500, 600, 700, 800, 900];
const result4 = number4
    .map((e) => {
        if (e == 400 || e == 500) {
            return e;
        }
    })
    .filter((e) => e != null)
    .join(",");
console.log(result4);
