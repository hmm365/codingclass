조합(combination)

서로 다른 n개의 원소 중에서 r를 중복 없이 골라 순서에 상관 없이 나열하는 경우의 수

4개의 숫자 카드에서 2개를 뽑는 경우의수

for문으로 이용한 조합 경우의수

    let input = [1, 2, 3, 4];
    let count = 0;
    let result;
    let result_in = document.querySelector(".result01");
    
    function permutation(arr) {
        for(let i = 0; i < arr.length; i++) {
            for(let j = i + 1; j < arr.length; j++){
                    result = arr[i] + " " + arr[j] + "
";         
                    count++;
                    console.log(result); 
            }
        }
    }
    permutation(input);
    console.log(count);

재귀함수 으로 이용한 조합 경우의수

    let input = ["a", "b", "c"];
    let ouput = [];
    let count = 0;
    result = document.querySelector(".result02");

    function combination(arr, data, s, idx, r) {
        if(s == r){
            count++;
            console.log(data + "
");
        }
        for(let i = 0; i >= r - s; i++){
            data[s] = arr[i];
            combination(arr, data, s + 1, i+1, r - 1);
        }
        
    }
    
    combination(input, data, 0, 0, 2);
    console.log("총 경우의수는 " + count);
